/* in oracle p_sequence_name

	start transaction;

	SET @sqlstmt = CONCAT('INSERT INTO  ',p_table_name,' (created_date) values (sysdate())');
    PREPARE erpstmt FROM @sqlstmt;

    DEALLOCATE PREPARE erpstmt;

	SELECT LAST_INSERT_ID() INTO p_sequence_no;
